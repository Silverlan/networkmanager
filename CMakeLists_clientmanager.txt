include(${CMAKE_SOURCE_DIR}/cmake/pr_common.cmake)

set(PROJ_NAME clientmanager)
pr_add_library(${PROJ_NAME} STATIC)

pr_add_dependency(${PROJ_NAME} networkmanager TARGET PUBLIC)
pr_add_external_dependency(${PROJ_NAME} boost HEADER_ONLY)
pr_add_external_dependency(${PROJ_NAME} boost_system LIBRARY LINK_ONLY)
pr_add_external_dependency(${PROJ_NAME} boost_date_time LIBRARY LINK_ONLY)
pr_add_external_dependency(${PROJ_NAME} boost_regex LIBRARY LINK_ONLY)

pr_add_headers(${PROJ_NAME} "include/")
pr_add_sources(${PROJ_NAME} "src/clientmanager/")

pr_add_compile_definitions(${PROJ_NAME} -DBOOST_ALL_DYN_LINK)

pr_finalize(${PROJ_NAME})
